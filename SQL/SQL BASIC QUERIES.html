CREATE TABLE orders (

    order_id VARCHAR(20) PRIMARY KEY,

    customer_id VARCHAR(20),

    product_id INT,

    order_date DATE,

    quantity INT,

    unit_price_inr DECIMAL(10,2),

    FOREIGN KEY (product_id)
        REFERENCES products(product_id)
);

📈 Intermediate SQL Analysis

13. What is the monthly sales trend?

SQL
SELECT
    DATE_FORMAT(order_date, '%Y-%m') AS Month,
    ROUND(SUM(total_sales_inr),2) AS Total_Revenue
FROM orders
GROUP BY Month
ORDER BY Month;

14. Which are the Top 10 Customers by Total Spending?

SQL
SELECT
    customer_id,
    ROUND(SUM(total_sales_inr),2) AS Total_Spending
FROM orders
GROUP BY customer_id
ORDER BY Total_Spending DESC
LIMIT 10;

15. Which are the Top 10 Best-Selling Products by Quantity?

SQL
SELECT
    p.product_name,
    SUM(o.quantity) AS Total_Quantity_Sold
FROM orders o
JOIN products p
ON o.product_id = p.product_id
GROUP BY p.product_name
ORDER BY Total_Quantity_Sold DESC
LIMIT 10;

16. Which states generated the highest revenue?

SQL
SELECT
    state,
    ROUND(SUM(total_sales_inr),2) AS Revenue
FROM orders
GROUP BY state
ORDER BY Revenue DESC;

17. What is the average spending per customer?

SQL
SELECT
    customer_id,
    ROUND(AVG(total_sales_inr),2) AS Average_Spending
FROM orders
GROUP BY customer_id
ORDER BY Average_Spending DESC;

18. Which customers placed more than one order?

SQL
SELECT
    customer_id,
    COUNT(order_id) AS Total_Orders
FROM orders
GROUP BY customer_id
HAVING COUNT(order_id) > 1
ORDER BY Total_Orders DESC;

19. Which payment method generated the highest revenue?

SQL
SELECT
    payment_method,
    ROUND(SUM(total_sales_inr),2) AS Revenue
FROM orders
GROUP BY payment_method
ORDER BY Revenue DESC;

20. What is the return rate by product category?

SQL
SELECT
    p.product_category,
    COUNT(CASE WHEN o.delivery_status = 'Returned' THEN 1 END) AS Returned_Orders,
    COUNT(*) AS Total_Orders,
    ROUND(
        COUNT(CASE WHEN o.delivery_status = 'Returned' THEN 1 END) * 100.0 / COUNT(*),
        2
    ) AS Return_Rate_Percentage
FROM orders o
JOIN products p
ON o.product_id = p.product_id
GROUP BY p.product_category
ORDER BY Return_Rate_Percentage DESC;

21. Which products have the highest average rating?

SQL
SELECT
    p.product_name,
    ROUND(AVG(o.review_rating),2) AS Average_Rating
FROM orders o
JOIN products p
ON o.product_id = p.product_id
GROUP BY p.product_name
ORDER BY Average_Rating DESC
LIMIT 10;

22. Which products have the lowest average rating?

SQL
SELECT
    p.product_name,
    ROUND(AVG(o.review_rating),2) AS Average_Rating
FROM orders o
JOIN products p
ON o.product_id = p.product_id
GROUP BY p.product_name
ORDER BY Average_Rating ASC
LIMIT 10;
